Recording Name Privacy

This script helps you hide your username while recording by replacing it locally with a randomized name.

-- Define the user's actual name and display name.
local player = game.Players.LocalPlayer
local realName = player.Name
local displayName = player.DisplayName

-- List of random names to select from.
local randomNames = {
    "SkyWalker", "PixelMage", "CodeCrafter", "GameChanger", "ShadowBlade",
    "SecretAgent", "MemeLord", "404NotFound", "Anonymous", "BananaKing",
    "TheChosenOne", "InvisibleInk", "DebugMaster", "OopsIDidItAgain"
}

-- Function to pick a random name from the list.
local function getRandomName()
    return randomNames[math.random(1, #randomNames)]
end

-- Function to replace occurrences of the username or display name.
local function replaceName(instance, newName)
    if instance:IsA("TextLabel") or instance:IsA("TextBox") or instance:IsA("TextButton") then
        -- Check and replace realName or displayName.
        if instance.Text:find(realName) then
            instance.Text = instance.Text:gsub(realName, newName)
        end
        if instance.Text:find(displayName) then
            instance.Text = instance.Text:gsub(displayName, newName)
        end
    end
end

-- Recursive function to traverse all descendants in the game and replace names.
local function traverseAndReplace(parent, newName)
    for _, child in pairs(parent:GetDescendants()) do
        replaceName(child, newName)
    end
end

-- Function to update the name every 2 minutes.
local function updateNamePeriodically()
    while true do
        local newName = getRandomName()
        traverseAndReplace(game, newName)
        wait(120) -- Wait for 2 minutes before changing again.
    end
end

-- Initial scan of the game with a random name.
traverseAndReplace(game, getRandomName())
-- Start the periodic name update.
task.spawn(updateNamePeriodically)
